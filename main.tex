\begin{center}
\Huge{Idea Proyecto final}
\end{center}
\justify
\large
{Juego de basketball 1 vs 1 de muñecos cabezones de dos jugadores, se puede escoger en una variedad de equipos y escenarios, el objetivo sera a llegar a una cantidad puntos seleccionada antes de que se acabe el tiempo. Y en el transcurso de la partida caeran objetos random que afectaran a los jugadores de manera positiva o negativa}
\section{\large{Metodos de Puntuacion: \normalfont{Si el tiro se realiza de la mitad de la cancha en adelante la cesta valdra 2 puntos y de la mitad hacia atras valdra 3 puntos}}}
\section{\large{\textbf{Modificadores Positivos:}}}
\subsection{Rayo: \normalfont{Mayor velocidad de movimiento.}}
\large
\subsection{Bebida energetica: \normalfont{Te aumenta el tamaño.}}
\subsection{2x: \normalfont{Todas las cestas valen el doble.}}
\section{\large{Modificadores Negativos:}}
\subsection{\large{Banana: \normalfont{te desliza hacia la cancha rival sin posibilidad de retroceder por 2 segundos}}}
\subsection{Cubo de Hielo:\normalfont{Te congela en tu lugar por 2 segundos} }
\subsection{Piernesitas de bebe: \normalfont{Te reduce el tamaño, dejandote sin posibilidad de bloquear tiros.}}
\section{\large{Modos de Ganar: \normalfont{Gana aquel que llegue a la puntacion seleccionada por el usuario o aquel que tenga la puntuacion mas alta al finalizar los 2 minutos}}}
